Key Idea 2 : 2. In Grasshopper, the default setting for list matching is "longest" list matching.

In Grasshopper, data is matched based on the list with the most elements. The last value of the shorter list is repeatedly matched to the extra elements in the longer list.

Write the c# script code by using Key Idea 1 and Key Idea 2.

C# Code Implementation:

private void RunScript(List<Line> x, object y, ref object A)
{
    // Divide the line by 2 square
    List<List<Point3d>> pt = new List<List<Point3d>>();
    int cnt = 2;

    for (int i = 0; i < x.Count; i++)
    {
        Curve tmp = x[i].ToNurbsCurve();
        var li = tmp.DivideByCount(cnt, true);
        List<Point3d> loc = new List<Point3d>();

        // Choose odd index and input the loc List
        for (int j = 1; j < li.Length; j += 2)
        {
            loc.Add(tmp.PointAt(li[j]));
        }

        pt.Add(loc);
        cnt *= 2;
    }

    // Initialize answer list
    List<List<Line>> ans = new List<List<Line>>();

    // Select some Points to others points
    for (int i = 0; i < x.Count - 1; i++)
    {
        List<Point3d> st = pt[i];
        List<Point3d> en = pt[i + 1];
        List<Line> liL = new List<Line>();

        for (int j = 0; j < st.Count; j++)
        {
            Point3d now = st[j];
            en.Sort((u, v) => u.DistanceTo(now).CompareTo(v.DistanceTo(now)));

            for (int k = 0; k < 2; k++)
            {
                liL.Add(new Line(now, en[k]));
            }
        }

        ans.Add(liL);
    }

    var ptLoc = makeDataTree(pt);
    var liLoc = makeDataTree(ans);
    A = liLoc;
}

#endregion
#region Additional

public static DataTree<T> makeDataTree<T>(List<List<T>> ret)
{
    DataTree<T> tree = new DataTree<T>();

    for (int i = 0; i < ret.Count; i++)
    {
        GH_Path path = new GH_Path(i);

        for (int j = 0; j < ret[i].Count; j++)
        {
            tree.Add(ret[i][j], path);
        }
    }

    return tree;
}

#endregion

C# script Code


C# Code Logic Description:

1. The process starts from the bottom and moves upwards, dividing lines into segments based on powers of two, and the positions of the division points are obtained.


2. Only the points at odd indices are selected because the final shape originates from the center.


3. Moving upwards again, the closest pairs of points are selected using sorting (with a time complexity of O(nlog(n))).


4. Lines are drawn between these selected points.
