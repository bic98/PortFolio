07_Variable offset curve.

Key Ideas:

1. Horizontal frame is useful for dynamic offset Curve

Key Idea 1 : 1. Horizontal frame is useful for dynamic offset Curve

The cross product of two lines is the normal vector of the surface formed by those two lines.

Therefore, the direction of all axes remains consistent with the perpendicular direction to the curve's tangent.

Write the c# script code by using Key Idea 1.

C# Code Implementation:

private void RunScript(Curve x, int y, Interval z, List<Plane> u, ref object A, ref object B)
{
    // Create horizontal planes along the curve and divide the curve by a given count
    var li = HorizontalPlanes(x, y, true);
    var t = x.DivideByCount(y, true);

    // Generate a range of intervals based on the input interval z
    var amp = RangeInterval(z.T0, z.T1, li.Count);

    // Get perpendicular planes along the curve (currently not used in this script)
    var k = PerpPlanes(x, y, true);

    // Lists to store the resulting curves and points
    List<Curve> ret = new List<Curve>();
    List<Point3d> L = new List<Point3d>();
    List<Point3d> R = new List<Point3d>();

    // Calculate points on the left and right side of the curve
    for (int i = 0; i < li.Count; i++)
    {
        Plane p = li[i];
        var v = p.YAxis;
        var tmp = MovePt(p.Origin, v, amp[i]);
        L.Add(tmp);
        tmp = MovePt(p.Origin, v, -amp[i]);
        R.Add(tmp);
    }

    // Create interpolated curves from the calculated points
    ret.Add(Curve.CreateInterpolatedCurve(L, 3));
    ret.Add(Curve.CreateInterpolatedCurve(R, 3));

    // Add arcs at the start and end of the curves
    ret.Add(new Arc(R[0], -x.TangentAt(t[0]), L[0]).ToNurbsCurve());
    ret.Add(new Arc(R[R.Count - 1], x.TangentAt(t[t.Length - 1]), L[L.Count - 1]).ToNurbsCurve());

    // Output the resulting curves and perpendicular planes
    A = ret;
    B = k;
}

// Generate a list of values evenly spaced between x and y
public static List<double> RangeInterval(double x, double y, int cnt)
{
    List<double> amp = new List<double>();
    if (cnt == 1) return amp;
    double gap = (y - x) / (cnt - 1);
    for (int i = 0; i < cnt; i++)
    {
        amp.Add(x + gap * i);
    }

    return amp;
}

// Create a list of horizontal planes along the curve
public static List<Plane> HorizontalPlanes(Curve x, int cnt, bool end)
{
    var t = x.DivideByCount(cnt, end);
    List<Plane> li = new List<Plane>();
    for (int i = 0; i < t.Length; i++)
    {
        var tan = x.TangentAt(t[i]);
        Plane p = new Plane(x.PointAt(t[i]), tan, Vector3d.CrossProduct(Vector3d.ZAxis, tan));
        li.Add(p);
    }

    return li;
}

// Convert a 2D list into a DataTree for output
public static DataTree<T> MakeDataTree2D<T>(List<List<T>> ret)
{
    DataTree<T> tree = new DataTree<T>();

    for (int i = 0; i < ret.Count; i++)
    {
        GH_Path path = new GH_Path(i);

        for (int j = 0; j < ret[i].Count; j++)
        {
            tree.Add(ret[i][j], path);
        }
    }

    return tree;
}

// Move a point by a vector scaled by a given amplitude
public Point3d MovePt(Point3d p, Vector3d v, double amp)
{
    v.Unitize();
    Vector3d newV = Vector3d.Multiply(v, amp);
    Point3d ans = Point3d.Add(p, newV);
    return ans;
}

C# Code Logic Description:

1. This code generates horizontal planes along a given curve and divides the curve into specified segments.


2. It calculates points on both sides of each horizontal plane and creates interpolated curves.


3. Arcs are added at the start and end of the generated curves to complete the shape.


4. The RangeInterval function generates evenly spaced values within a specified range.


5. The HorizontalPlanes function creates planes along the curve and returns them as a list.


6. The MovePt function moves a given point using a vector.
